## Overview

The `CountryContext` is a React context that provides global state for countries, loading status, and search history. It's consumed via the `useCountry` hook and provided by the `CountryProvider` component.

**Location:** `src/context/CountryContext.jsx`

## Context Definition

```javascript
import { createContext, useContext } from "react";

export const CountryContext = createContext();

export const useCountry = () => useContext(CountryContext);
```

## Context Shape

The context provides the following values:

```typescript
type CountryContextValue = {
  isLoading: boolean;
  setIsLoading: (loading: boolean) => void;
  countries: Country[];
  setCountries: (countries: Country[]) => void;
  history: HistoryItem[];
  setHistory: (history: HistoryItem[] | ((prev: HistoryItem[]) => HistoryItem[])) => void;
}
```

### Setter Functions

<ResponseField name="setIsLoading" type="function">
  Updates the loading state
  
  **Signature:** `(loading: boolean) => void`
  
  **Example:**
  ```javascript
  setIsLoading(true);  // Show loading indicator
  setIsLoading(false); // Hide loading indicator
  ```
</ResponseField>

<ResponseField name="setCountries" type="function">
  Updates the countries array
  
  **Signature:** `(countries: array) => void`
  
  **Example:**
  ```javascript
  setCountries(apiResponse.data);
  ```
</ResponseField>

<ResponseField name="setHistory" type="function">
  Updates the history array (syncs to localStorage)
  
  **Signature:** `(history: array | function) => void`
  
  **Examples:**
  ```javascript
  // Direct update
  setHistory([newItem, ...oldHistory]);
  
  // Functional update (recommended)
  setHistory((prev) => {
    const filtered = prev.filter(item => item.name !== newItem.name);
    return [newItem, ...filtered];
  });
  ```
</ResponseField>

## Usage

### Consuming the Context

Use the `useCountry` hook to access the context:

```javascript
import { useCountry } from "../context/CountryContext";

function MyComponent() {
  const { countries, isLoading, history } = useCountry();
  
  if (isLoading) {
    return <div>Loading...</div>;
  }
  
  return (
    <div>
      <p>{countries.length} countries loaded</p>
      <p>{history.length} items in history</p>
    </div>
  );
}
```

### Provider Setup

The context is provided by `CountryProvider`:

```javascript
import { CountryProvider } from "./context/CountryProvider";
import App from "./App";

function Root() {
  return (
    <CountryProvider>
      <App />
    </CountryProvider>
  );
}
```

## Real-World Examples

### Example 1: Search Component

```javascript src/components/BuscadorCountry.jsx
import { useCountry } from "../context/CountryContext";

function BuscarCountry() {
  const { countries, setIsLoading } = useCountry();
  const navigate = useNavigate();
  
  const handleSearch = (searchTerm) => {
    const country = countries.find(c => 
      c.translations.spa.common.toLowerCase() === searchTerm.toLowerCase()
    );
    
    if (country) {
      setIsLoading(true);
      navigate(`/country/${country.translations.spa.common.toLowerCase()}`);
    }
  };
  
  return (
    <input
      type="text"
      placeholder="Search countries..."
      onChange={(e) => handleSearch(e.target.value)}
    />
  );
}
```

### Example 3: Adding to History

```javascript src/hooks/useCountryInfo.js
import { useCountry } from "../context/CountryContext";

export const useCountryInfo = (name) => {
  const { setHistory } = useCountry();
  
  const addToHistory = (countryData) => {
    const log = {
      name: countryData.translations.spa.common,
      flags: countryData.flags,
    };
    
    setHistory((prev) => {
      // Don't duplicate if it's already the most recent
      if (prev[0]?.name === log.name) return prev;
      
      // Remove any existing entry for this country
      const filtered = prev.filter(c => c.name !== log.name);
      
      // Add to the beginning
      return [log, ...filtered];
    });
  };
  
  return { addToHistory };
};
```

## Context Provider Implementation

See the full provider implementation:

```javascript src/context/CountryProvider.jsx
import { useState, useEffect } from "react";
import { CountryContext } from "./CountryContext";
import { getAll } from "../api/country";

export const CountryProvider = ({ children }) => {
  const [isLoading, setIsLoading] = useState(true);
  const [countries, setCountries] = useState([]);
  const [history, setHistory] = useState(() => {
    return JSON.parse(localStorage.getItem("history")) || [];
  });

  useEffect(() => {
    const fetchCountries = async () => {
      try {
        const res = await getAll();
        setCountries(res.data);
      } catch (error) {
        console.error("Error al obtener países:", error);
      } finally {
        setIsLoading(false);
      }
    };
    fetchCountries();
  }, []);

  useEffect(() => {
    localStorage.setItem("history", JSON.stringify(history));
  }, [history]);

  return (
    <CountryContext.Provider
      value={{
        isLoading,
        setIsLoading,
        countries,
        setCountries,
        history,
        setHistory,
      }}
    >
      {children}
    </CountryContext.Provider>
  );
};
```

## Type Safety

For TypeScript projects, define the context type:

```typescript
interface Country {
  name: {
    common: string;
    official: string;
  };
  capital: string[];
  flags: {
    svg: string;
    png: string;
    alt: string;
  };
  population: number;
  cca3: string;
  translations: {
    spa: {
      common: string;
      official: string;
    };
  };
}

interface HistoryItem {
  name: string;
  flags: {
    svg: string;
    alt: string;
  };
}

interface CountryContextValue {
  isLoading: boolean;
  setIsLoading: (loading: boolean) => void;
  countries: Country[];
  setCountries: (countries: Country[]) => void;
  history: HistoryItem[];
  setHistory: (history: HistoryItem[] | ((prev: HistoryItem[]) => HistoryItem[])) => void;
}

export const CountryContext = createContext<CountryContextValue | undefined>(undefined);
```

## Best Practices

### 1. Always Use the Hook

```javascript
// ✅ Good
import { useCountry } from "../context/CountryContext";
const { countries } = useCountry();

// ❌ Bad
import { CountryContext } from "../context/CountryContext";
const value = useContext(CountryContext);
```

### 2. Destructure Only What You Need

```javascript
// ✅ Good - only causes re-renders when countries changes
const { countries } = useCountry();

// ⚠️ Okay but less optimal
const context = useCountry();
return <div>{context.countries.length}</div>;
```

### 3. Use Functional Updates for History

```javascript
// ✅ Good - uses previous state
setHistory((prev) => [newItem, ...prev]);

// ❌ Bad - may use stale state
setHistory([newItem, ...history]);
```
