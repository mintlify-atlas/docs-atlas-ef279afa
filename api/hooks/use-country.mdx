## Overview

The `useCountry` hook provides access to the global country context, including the list of all countries, loading state, and search history. It's a convenience wrapper around React's `useContext` for the `CountryContext`.

**Location:** `src/context/CountryContext.jsx`

## Hook Signature

```javascript
export const useCountry = () => useContext(CountryContext)
```

## Parameters

This hook does not accept any parameters.

## Return Value

The hook returns the complete `CountryContext` value object:

<ResponseField name="isLoading" type="boolean">
  Global loading state for async operations (country fetching, navigation)
</ResponseField>

<ResponseField name="setIsLoading" type="function">
  Function to update the loading state
  
  **Signature:** `(loading: boolean) => void`
</ResponseField>

<ResponseField name="countries" type="array">
  Array of all countries loaded from the REST Countries API
  
  <Expandable title="Country Object Structure">
    <ResponseField name="name.common" type="string">
      Common English name
    </ResponseField>
    
    <ResponseField name="capital" type="array">
      Array of capital city names
    </ResponseField>
    
    <ResponseField name="flags" type="object">
      Flag image URLs and alt text
    </ResponseField>
    
    <ResponseField name="population" type="number">
      Population count
    </ResponseField>
    
    <ResponseField name="cca3" type="string">
      ISO 3166-1 alpha-3 code
    </ResponseField>
    
    <ResponseField name="translations.spa" type="object">
      Spanish name translations
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="setCountries" type="function">
  Function to update the countries array
  
  **Signature:** `(countries: array) => void`
</ResponseField>

<ResponseField name="history" type="array">
  Array of recently viewed countries
  
  <Expandable title="History Item Structure">
    <ResponseField name="name" type="string">
      Spanish common name of the country
    </ResponseField>
    
    <ResponseField name="flags" type="object">
      Flag image data (svg, alt)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="setHistory" type="function">
  Function to update the history array
  
  **Signature:** `(history: array | function) => void`
  
  Accepts either a new array or an updater function
</ResponseField>

## Usage

```jsx
import { useCountry } from "../context/CountryContext";

function CountryList() {
  const { countries, isLoading } = useCountry();
  
  if (isLoading) {
    return <div>Loading countries...</div>;
  }
  
  return (
    <ul>
      {countries.map((country) => (
        <li key={country.cca3}>
          {country.translations.spa.common}
        </li>
      ))}
    </ul>
  );
}
```

## Implementation

```javascript src/context/CountryContext.jsx
import { createContext, useContext } from "react";

export const CountryContext = createContext();

export const useCountry = () => useContext(CountryContext);
```

## Context Provider

The context is provided by `CountryProvider`:

```javascript src/context/CountryProvider.jsx
<CountryContext.Provider
  value={{
    isLoading,
    setIsLoading,
    countries,
    setCountries,
    history,
    setHistory,
  }}
>
  {children}
</CountryContext.Provider>
```

## Common Use Cases

### 1. Accessing All Countries

```jsx
import { useCountry } from "../context/CountryContext";

function BuscadorComponent() {
  const { countries } = useCountry();
  
  return (
    <div>
      <p>Datos disponibles de <b>{`+${countries.length} paises.`}</b></p>
    </div>
  );
}
```

### 2. Managing Loading State

```jsx
import { useCountry } from "../context/CountryContext";
import { useNavigate } from "react-router-dom";

function SearchForm() {
  const { setIsLoading } = useCountry();
  const navigate = useNavigate();
  
  const handleSearch = (countryName) => {
    setIsLoading(true);
    navigate(`/country/${countryName}`);
  };
  
  return (
    <button onClick={() => handleSearch("españa")}>Search Spain</button>
  );
}
```

### 3. Accessing Search History

```jsx
import { useCountry } from "../context/CountryContext";

function HistoryComponent() {
  const { history } = useCountry();
  
  return (
    <div>
      <h4>Historial de busquedas ({history.length})</h4>
      <ul>
        {history.map((country, index) => (
          <li key={index}>
            <img src={country.flags.svg} alt={country.flags.alt} />
            <p>{country.name}</p>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

### 4. Updating History

```jsx
import { useCountry } from "../context/CountryContext";

function useAddToHistory() {
  const { setHistory } = useCountry();
  
  const addCountry = (countryData) => {
    const log = {
      name: countryData.translations.spa.common,
      flags: countryData.flags,
    };
    
    setHistory((prev) => {
      const lastOne = prev[0];
      if (lastOne && lastOne.name === log.name) return prev;
      const prevSet = prev.filter((c) => log.name !== c.name);
      return [log, ...prevSet];
    });
  };
  
  return addCountry;
}
```

### 5. Searching Countries

```jsx
import { useCountry } from "../context/CountryContext";
import { normalize } from "../utils/libs";

function useSearchCountry() {
  const { countries } = useCountry();
  
  const searchByName = (name) => {
    return countries.find((country) => {
      const nameSpa = country.translations.spa.common;
      const nameEng = country.name.common;
      
      return (
        normalize(nameSpa) === normalize(name) ||
        normalize(nameEng) === normalize(name)
      );
    });
  };
  
  return searchByName;
}
```

## Context Value Shape

```typescript
type CountryContextValue = {
  isLoading: boolean;
  setIsLoading: (loading: boolean) => void;
  countries: Array<{
    name: { common: string; official: string };
    capital: string[];
    flags: { svg: string; png: string; alt: string };
    population: number;
    cca3: string;
    translations: {
      spa: { common: string; official: string };
    };
  }>;
  setCountries: (countries: any[]) => void;
  history: Array<{
    name: string;
    flags: { svg: string; alt: string };
  }>;
  setHistory: (history: any[] | ((prev: any[]) => any[])) => void;
};
```

## Requirements

### Provider Requirement

This hook **must** be used within a `CountryProvider`:

```jsx
import { CountryProvider } from "./context/CountryProvider";
import { App } from "./App";

function Root() {
  return (
    <CountryProvider>
      <App />
    </CountryProvider>
  );
}
```

### Error Handling

If used outside a provider, React will throw an error:

```
Error: useCountry must be used within a CountryProvider
```

To handle this gracefully:

```jsx
export const useCountry = () => {
  const context = useContext(CountryContext);
  
  if (context === undefined) {
    throw new Error('useCountry must be used within a CountryProvider');
  }
  
  return context;
};
```
